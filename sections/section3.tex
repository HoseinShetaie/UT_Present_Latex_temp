\begin{frame}{Vlasov-Poisson equation}
Vlasov-Poisson equation:

        \begin{equation*}\begin{gathered} \frac{{\partial f}}{{\partial t}} + {\mathbf{v}} \cdot \nabla f + \frac{q}{m}\left({\mathbf{E}} + {\mathbf{v}} \times {\mathbf{B}}\right) \cdot \frac{{\partial f}}{{\partial {\mathbf{v}}}} = 0 \\ \nabla \cdot {\mathbf{E}}{\,\boldsymbol{ = }}\,\frac{\rho }{\varepsilon } \\ \end{gathered} \end{equation*}
        \pause
Density:	

        \begin{equation*}\begin{gathered} \rho = e(n_e-Z_i n_i)  \\ {n_{\text{e}}} = \int\limits_{ - \infty }^{ + \infty } f\left(t,x,v\right){\text{d}}v \\ \end{gathered} \end{equation*}
        \pause
1D Vlasov-Poisson equation:

			\begin{align*} & \frac{{\partial f}}{{\partial t}} + v\frac{{\partial f}}{{\partial x}} + \frac{{eE}}{{{m_{\text{e}}}}}\frac{{\partial f}}{{\partial v}} = 0 \nonumber\\ & \frac{{\partial E}}{{\partial x}} = \frac{{e\left({n_{\text{e}}} - {n_i}\right)}}{\varepsilon }\end{align*}
			

\end{frame}

\begin{frame}{Vlasov-Poisson equation}
Normalization:
    \begin{equation*}x = \frac{{{x^*}}}{{{\lambda _{\text{D}}}}},\,v = \frac{{{v^*}}}{{{v_{{\text{th}}}}}},\,t = \frac{{{t^*}}}{{\omega _{\text{p}}^{ - 1}}},\,m = \frac{{{m^*}}}{{{m_{\text{e}}}}},\,q = \frac{{{q^*}}}{e},\,E = {E^*}\frac{{e{\lambda _{\text{D}}}}}{{k{T_{\text{e}}}}}\end{equation*}

\pause
Normalized 1D Vlasov-Poisson equation:
		
	\begin{align*}
         & \frac{{\partial f}}{{\partial t}} + v\frac{{\partial f}}{{\partial x}} - {\textrm E}\frac{{\partial f}}{{\partial v}} = 0 \nonumber\\ & \frac{{\partial {\textrm E}}}{{\partial x}} = \int\limits_{ - \infty }^{ + \infty } f\left(t,x,v\right){\text{d}}v - 1
    \end{align*}

\end{frame}
